  <!-- JS: Smooth effects, carousels and spray animation trigger -->
  <script>
    // YEAR
    document.getElementById('year').textContent = new Date().getFullYear();

    /* Simple carousel controls: buttons with data-target */
    document.querySelectorAll('.btn').forEach(btn => {
      btn.addEventListener('click', () => {
        const trackId = btn.getAttribute('data-target');
        const dir = parseInt(btn.getAttribute('data-dir'), 10);
        const track = document.getElementById(trackId);
        if(!track) return;
        const itemWidth = track.querySelector('.carousel-item')?.offsetWidth || 260;
        // compute visible count based on container width (approx)
        track.style.transform = translateX(${(track._pos || 0) - dir * itemWidth}px);
        track._pos = (track._pos || 0) - dir * itemWidth;
        // clamp boundaries
        const maxShift = 0;
        const minShift = -Math.max(0, track.scrollWidth - track.parentElement.offsetWidth);
        if(track._pos > maxShift) { track._pos = maxShift; track.style.transform = translateX(${maxShift}px); }
        if(track._pos < minShift) { track._pos = minShift; track.style.transform = translateX(${minShift}px); }
      });
    });

    /* Show spray animation when section enters viewport */
    const spray = document.getElementById('sprayOverlay');
    const observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if(entry.isIntersecting){
          // show spray when new section is visible
          spray.classList.remove('show');
          void spray.offsetWidth; // restart animation
          spray.classList.add('show');
        }
      });
    }, { threshold: 0.35 });

    document.querySelectorAll('main section').forEach(sec => observer.observe(sec));

    /* Smooth scroll + highlight nav */
    const navLinks = document.querySelectorAll('nav a');
    function onScrollHighlight(){
      const scrollPos = window.scrollY + window.innerHeight/3;
      document.querySelectorAll('main section').forEach(sec => {
        const top = sec.offsetTop;
        const bottom = top + sec.offsetHeight;
        const id = sec.getAttribute('id');
        const link = document.querySelector('nav a[href="#'+id+'"]');
        if(scrollPos >= top && scrollPos < bottom){
          navLinks.forEach(a => a.classList.remove('active'));
          if(link) link.classList.add('active');
        }
      });
    }
    window.addEventListener('scroll', onScrollHighlight);
    onScrollHighlight();

    /* Form handling: show friendly message on submit (Formspree will handle actual delivery) */
    const contactForm = document.getElementById('contactForm');
    const formStatus = document.getElementById('formStatus');

    contactForm.addEventListener('submit', function(e){
      // optionally prevent default and do AJAX submit to give instant feedback
      e.preventDefault();
      formStatus.textContent = 'Enviando...';
      const data = new FormData(contactForm);
      const action = contactForm.getAttribute('action');
      fetch(action, {
        method: 'POST',
        body: data,
        headers: { 'Accept': 'application/json' }
      }).then(response => {
        if(response.ok){
          formStatus.textContent = '¡Mensaje enviado! Te responderé pronto.';
          contactForm.reset();
        } else {
          response.json().then(r => {
            formStatus.textContent = r?.error || 'Error al enviar. Intenta de nuevo.';
          }).catch(()=> formStatus.textContent = 'Error al enviar. Intenta de nuevo.');
        }
      }).catch(() => formStatus.textContent = 'Error de red. Intenta de nuevo más tarde.');
    });

    /* Optional: lazy-loading images */
    document.querySelectorAll('img').forEach(img => {
      if('loading' in HTMLImageElement.prototype) img.loading = 'lazy';
    });

    /* Add subtle parallax to decor elements for depth (optional) */
    window.addEventListener('mousemove', (e) => {
      const w = window.innerWidth, h = window.innerHeight;
      const px = (e.clientX - w/2) / w;
      const py = (e.clientY - h/2) / h;
      document.querySelectorAll('.decor svg').forEach((svg, i) => {
        const depth = (i % 2 === 0) ? 6 : 12;
        svg.style.transform = translate3d(${px*depth}px, ${py*depth}px, 0);
      });
    });

    /* Accessibility: allow keyboard navigation for carousels (left/right) */
    window.addEventListener('keydown', (e) => {
      if(e.key === 'ArrowLeft' || e.key === 'ArrowRight'){
        // apply to visible carousel (first one in viewport)
        const carousels = document.querySelectorAll('.carousel .carousel-track');
        for(const c of carousels){
          const rect = c.getBoundingClientRect();
          if(rect.top < window.innerHeight && rect.bottom > 0){
            const dir = (e.key === 'ArrowLeft') ? -1 : 1;
            // find control for this track
            const id = c.id;
            const btn = document.querySelector(.btn[data-target='${id}'][data-dir='${dir}']);
            if(btn){ btn.click(); }
            break;
          }
        }
      }
    });
  </script>
